class SwissRollDataset(Dataset):
    def __init__(self, n_samples, noise=0.1, train=True):
        # Generate data
        X, t = make_swiss_roll(n_samples=n_samples, noise=noise)
        # Convert to tensor
        self.data = torch.FloatTensor(X)
        
        # Split into train/test (80/20)
        if train:
            self.data = self.data[:int(0.8 * n_samples)]
        else:
            self.data = self.data[int(0.8 * n_samples):]

    def __len__(self):
        return len(self.data)

    def __getitem__(self, idx):
        return self.data[idx]

# score_network takes input of Swiss Roll image dimension and returns the output of
# the same size, socre network usually follows the structure of U-Net

device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
class ResidualBlock(nn.Module):
    def __init__(self, dim):
        super(ResidualBlock, self).__init__()
        self.net = nn.Sequential(
            nn.Linear(dim, dim),
            nn.BatchNorm1d(dim),
            nn.Softplus(),
            nn.Linear(dim, dim),
            nn.BatchNorm1d(dim),
            nn.Softplus()
        )

    def forward(self, x):
        return self.net(x) + x

class ScoreNet(nn.Module):
    def __init__(self, input_dim=3, hidden_dim=256, n_blocks=2):
        super(ScoreNet, self).__init__()
        
        # Input projection
        self.input_layer = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.BatchNorm1d(hidden_dim),
            nn.Softplus()
        )
        
        # Residual blocks
        self.residual_blocks = nn.ModuleList([
            ResidualBlock(hidden_dim) for _ in range(n_blocks)
        ])
        
        # Hidden layers
        self.hidden_layers = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim),
            nn.BatchNorm1d(hidden_dim),
            nn.Softplus(),
            nn.Linear(hidden_dim, hidden_dim//2),
            nn.BatchNorm1d(hidden_dim//2),
            nn.Softplus()
        )
        
        # Output projection
        self.output_layer = nn.Linear(hidden_dim//2, input_dim)

    def forward(self, x):
        h = self.input_layer(x)
        for block in self.residual_blocks:
            h = block(h)
        h = self.hidden_layers(h)
        score = self.output_layer(h)
        return score

# Initialize network
score_net = ScoreNet(input_dim=3, hidden_dim=256, n_blocks=4)

# Implement the sliced score matching function, as illustrated in sldes 26-28
# of lectur 4 of Teaching Staff Lecture Slides
def calc_loss(score_net, x):
    """
    Compute sliced score matching loss with gradient tracking enabled
    """
    # Enable gradients for input
    x.requires_grad_(True)
    
    # Sample random vectors
    v = torch.randn_like(x)
    v = v / torch.norm(v, dim=1, keepdim=True)
    
    # Get score estimates
    scores = score_net(x)
    
    # Compute v^T * score(x)
    v_score = torch.sum(v * scores, dim=1)
    
    # Square term
    square_term = 0.5 * v_score.pow(2)
    
    # Compute gradient term efficiently
    grad_v_score = torch.autograd.grad(
        v_score.sum(), x,
        create_graph=True,
        retain_graph=True
    )[0]
    grad_term = torch.sum(v * grad_v_score, dim=1)
    
    # Full loss
    loss = (square_term + grad_term).mean()
    
    return loss

# define the ULA algorithm:

def generate_samples(score_net, n_samples=1000, eps=0.001, T=50):
    """
    Generate samples using Unadjusted Langevin Algorithm (ULA)
    """
    # Initialize random samples
    x = torch.randn(n_samples, 3)
    eps_tensor = torch.tensor(eps)
    
    # ULA iterations
    for t in range(T):
        # Get score estimate
        score = score_net(x)
        
        # Update samples using Langevin dynamics
        noise = torch.randn_like(x)
        x = x + eps * score + torch.sqrt(torch.tensor(2.0) * eps) * noise
    
    return x
